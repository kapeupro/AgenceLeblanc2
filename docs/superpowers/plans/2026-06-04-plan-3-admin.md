# Agence Leblanc — Plan 3 : Admin back-office

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Prérequis :** Plans 1 et 2 terminés.

**Goal:** Construire le back-office admin : login JWT, dashboard, CRUD annonces avec upload de photos, gestion des messages de contact, et gestion de l'équipe.

**Architecture:** Routes TanStack Router sous `/admin/*`. Le layout `/admin/__root.tsx` vérifie `GET /api/auth/me` avant chaque navigation — si non connecté, redirect vers `/admin/login`. Les mutations utilisent TanStack Query `useMutation`. L'upload de photos envoie un `FormData` vers `POST /api/upload` puis attache l'URL au bien.

**Tech Stack:** Même que Plan 2 + `@tanstack/react-table` v8 pour les tableaux.

```tsx
import { createFileRoute, useNavigate } from '@tanstack/react-router';
import { useState } from 'react';
import { api } from '../../lib/api';

export const Route = createFileRoute('/admin/login')({ component: LoginPage });

function LoginPage() {
  const navigate = useNavigate();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');
  const [loading, setLoading] = useState(false);

  const submit = async (e: React.FormEvent) => {
    e.preventDefault();
    setError(''); setLoading(true);
    const { data, error: err } = await api.api.auth.login.post({ email, password });
    setLoading(false);
    if (err || !(data as any)?.ok) { setError('Email ou mot de passe incorrect'); return; }
    navigate({ to: '/admin' });
  };

  return (
    <div className="min-h-screen bg-navy flex items-center justify-center p-6">
      <div className="bg-white rounded-[20px] p-10 w-full max-w-[420px] shadow-[var(--shadow-card)]">

          <div className="text-muted mt-1.5 text-[15px]">Espace administration</div>
        </div>
        <form onSubmit={submit}>
          <div className="mb-4.5">
            <label className="block font-semibold text-[14.5px] text-head mb-2">Email</label>
            <input type="email" required value={email} onChange={e => setEmail(e.target.value)}
              className="w-full border border-[#dfe3ec] rounded-[12px] px-4 py-3.5 text-[15px] focus:outline-none focus:border-navy" />

          <div className="mb-6">

          </div>
          {error && <p className="text-coral text-[14px] mb-4 text-center font-medium">{error}</p>}
          <button type="submit" disabled={loading}
            className="w-full bg-navy text-white rounded-[12px] py-4 font-semibold text-[15px] hover:bg-navy-deep transition-colors disabled:opacity-60">
            {loading ? 'Connexion…' : 'Se connecter'}
          </button>
        </form>
      </div>
    </div>
  );
}
```

```tsx
import { createFileRoute, Link } from '@tanstack/react-router';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { api } from '../../../lib/api';

export const Route = createFileRoute('/admin/biens/')({ component: AdminBiensPage });

function fmtPrice(cents: number) {
  return (cents / 100).toLocaleString('fr-FR', { maximumFractionDigits: 0 }) + ' €';
}

const TYPE_LABEL: Record<string, string> = {
  maison: 'Maison', appartement: 'Appt', terrain: 'Terrain',
  local: 'Local', immeuble: 'Immeuble', cave: 'Cave',
};

function AdminBiensPage() {

  const deleteMutation = useMutation({
    mutationFn: (id: string) => api.api.properties({ id }).delete(),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['admin-properties'] }),

        <div className="bg-white rounded-[16px] border border-line overflow-hidden shadow-[var(--shadow-soft)]">
          <table className="w-full text-[14px]">

                  <td className="px-5 py-3.5 font-semibold text-head whitespace-nowrap">{fmtPrice(p.price)}</td>

                    <span className={`inline-flex items-center px-2.5 py-1 rounded-full text-[12px] font-semibold

                        className="text-coral font-semibold hover:underline text-[13.5px] disabled:opacity-50"
                        disabled={deleteMutation.isPending}>
                        Supprimer
                      </button>
                    </div>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}
    </div>
  );
}
```

```tsx
function toFormData(p: Property): FormData {
  return {
    slug: p.slug, type: p.type, title: p.title, city: p.city,
    price: String(p.price / 100), status: p.status, exclusive: p.exclusive,
    description: p.description ?? '', heat: p.heat ?? '',
    beds: p.beds != null ? String(p.beds) : '', rooms: p.rooms != null ? String(p.rooms) : '',
    area: p.area != null ? String(p.area) : '', land: p.land != null ? String(p.land) : '',
    year: p.year != null ? String(p.year) : '',
    dpeValue: p.dpeValue != null ? String(p.dpeValue) : '', dpeClass: p.dpeClass ?? '',
    gesValue: p.gesValue != null ? String(p.gesValue) : '', gesClass: p.gesClass ?? '',
    energyCost: p.energyCost ?? '', hasVideo: p.hasVideo, hasTour: p.hasTour,
    agentId: p.agentId ?? '',

}

function fromFormData(f: FormData) {
  return {
    slug: f.slug, type: f.type, title: f.title, city: f.city,
    price: Math.round(Number(f.price) * 100),
    status: f.status, exclusive: f.exclusive,
    description: f.description || null, heat: f.heat || null,
    beds: f.beds ? Number(f.beds) : null, rooms: f.rooms ? Number(f.rooms) : null,
    area: f.area ? Number(f.area) : null, land: f.land ? Number(f.land) : null,
    year: f.year ? Number(f.year) : null,
    dpeValue: f.dpeValue ? Number(f.dpeValue) : null, dpeClass: f.dpeClass || null,
    gesValue: f.gesValue ? Number(f.gesValue) : null, gesClass: f.gesClass || null,
    energyCost: f.energyCost || null, hasVideo: f.hasVideo, hasTour: f.hasTour,
    agentId: f.agentId || null, features: [], near: [],

}

const EMPTY: FormData = {

  const [form, setForm] = useState<FormData>(initial ?? EMPTY);
```

```tsx
import { createFileRoute } from '@tanstack/react-router';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { useState } from 'react';
import { api } from '../../lib/api';

export const Route = createFileRoute('/admin/equipe')({ component: EquipePage });

function EquipePage() {
  const qc = useQueryClient();
  const { data: team = [] } = useQuery({
    queryKey: ['team'],
    queryFn: async () => { const { data } = await api.api.team.get(); return (data ?? []) as any[]; },
  });

  const [form, setForm] = useState({ name: '', role: '', photoUrl: '', displayOrder: '0' });
  const set = (k: string) => (e: React.ChangeEvent<HTMLInputElement>) => setForm(f => ({ ...f, [k]: e.target.value }));

  const create = useMutation({
    mutationFn: () => api.api.team.post({ name: form.name, role: form.role, photoUrl: form.photoUrl || null, displayOrder: Number(form.displayOrder) }),
    onSuccess: () => { qc.invalidateQueries({ queryKey: ['team'] }); setForm({ name: '', role: '', photoUrl: '', displayOrder: '0' }); },
  });

  const remove = useMutation({
    mutationFn: (id: string) => api.api.team({ id }).delete(),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['team'] }),
  });

  return (
    <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
      {/* Liste */}
      <div>
        <h3 className="text-[18px] font-bold text-head mb-4">Membres ({(team as any[]).length})</h3>
        <div className="space-y-3">
          {(team as any[]).map((m: any) => (
            <div key={m.id} className="bg-white rounded-[12px] border border-line p-4 flex items-center gap-4 shadow-[var(--shadow-soft)]">
              <div className="w-12 h-12 rounded-full bg-gray-bg overflow-hidden flex-none">
                {m.photoUrl ? <img src={m.photoUrl} alt="" className="w-full h-full object-cover" /> : <div className="w-full h-full flex items-center justify-center text-muted text-lg">👤</div>}
              </div>
              <div className="flex-1 min-w-0">
                <b className="block text-head text-[15px]">{m.name}</b>
                <span className="text-muted text-[13px]">{m.role}</span>
              </div>
              <button onClick={() => { if (confirm(`Supprimer ${m.name} ?`)) remove.mutate(m.id); }}
                className="text-coral text-[13px] font-semibold hover:underline">Supprimer</button>
            </div>
          ))}
        </div>
      </div>

      {/* Ajout */}
      <div>
        <h3 className="text-[18px] font-bold text-head mb-4">Ajouter un membre</h3>
        <form className="bg-white rounded-[14px] border border-line p-6 shadow-[var(--shadow-soft)]"
          onSubmit={e => { e.preventDefault(); create.mutate(); }}>
          {[['Nom complet *', 'name', true], ['Rôle *', 'role', true], ['URL photo', 'photoUrl', false], ['Ordre d\'affichage', 'displayOrder', false]].map(([label, key, req]) => (
            <div key={key as string} className="mb-4">
              <label className="block font-semibold text-[14px] text-head mb-1.5">{label as string}</label>
              <input required={!!req} value={(form as any)[key as string]} onChange={set(key as string)}
                type={key === 'displayOrder' ? 'number' : 'text'}
                className="w-full border border-[#dfe3ec] rounded-[12px] px-4 py-3 text-[14.5px] focus:outline-none focus:border-navy" />
            </div>
          ))}
          <button type="submit" disabled={create.isPending}
            className="bg-navy text-white rounded-[12px] px-6 py-3 font-semibold text-[14.5px] hover:bg-navy-deep transition-colors disabled:opacity-60">
            {create.isPending ? 'Ajout…' : '+ Ajouter'}
          </button>
        </form>
      </div>
    </div>
  );
}
```
